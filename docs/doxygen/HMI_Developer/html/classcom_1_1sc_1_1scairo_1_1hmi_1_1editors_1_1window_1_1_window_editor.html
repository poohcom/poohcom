public void addView(HMIDesignView view) {
    int oldCount = views.size();
    views.add(view);
    view.addPropertyChangeListener(HMIDesignView.FILE_PROPERTY, this);
    firePropertyChange(VIEW_COUNT_PROPERTY, oldCount, views.size());
}

public void removeView(HMIDesignView view) {
    hideView(view);
    int oldCount = views.size();
    views.remove(view.getFile());
    view.removePropertyChangeListener(HMIDesignView.FILE_PROPERTY, this);
    view.dispose();

    firePropertyChange(VIEW_COUNT_PROPERTY, oldCount, views.size());

    setActiveView(views.getFirst());

setActiveView(views.getLast()); }
