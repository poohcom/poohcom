private final void setDeveloper(Developer owner) { developer = owner; }

private CommandBar createCommandBar(String key) { CommandBar createdBar = null;

if (FILE_TOOL_KEY.equals(key)) { createdBar = createFileCommandBar(); } else if (EDIT_TOOL_KEY.equals(key)) { createdBar = createEditCommandBar(); } else if (PROJECT_TOOL_KEY.equals(key)) { createdBar = createProjectCommandBar(); } else if (UTILITY_TOOL_KEY.equals(key)) { createdBar = createUtilityCommandBar(); } else if (MENU_BAR_KEY.equals(key)) { createdBar = createMenuCommandBar(getDeveloper()); } else { System.out.println(key + " is not exist in the command bar factory.."); //$NON-NLS-1$ createdBar = null; }

if (null != createdBar) { commandBarTable.put(key, createdBar); } return createdBar; }
