private void setConversionComboBox(int select) {
    if(currentTag instanceof DigitalTag) {
        if(select == ITagConstants.CONVERSION_REVERSE) {
            conversionComboBox.setSelectedIndex(1);
        }
    }
    else if(currentTag instanceof AnalogTag) {
        if(select == ITagConstants.CONVERSION_MINMAX) {
            conversionComboBox.setSelectedIndex(1);
        }
        else if(select == ITagConstants.CONVERSION_FOMULA) {
            conversionComboBox.setSelectedIndex(2);
        }
        else if(select == ITagConstants.CONVERSION_SCALE_OFFSET) {
            conversionComboBox.setSelectedIndex(3);
        }
    } 
    if(select == ITagConstants.CONVERSION_NONE) {
        conversionComboBox.setSelectedIndex(0);
    }
}

private int getConversionComboBox(Tag tag) {
    int select = ITagConstants.CONVERSION_NONE;
    if(tag instanceof DigitalTag) {
        if(conversionComboBox.getSelectedIndex() == 1) {
            return select = ITagConstants.CONVERSION_REVERSE;
        }
    }
    else if(tag instanceof AnalogTag) {
        if(conversionComboBox.getSelectedIndex() == 1) {
            return select = ITagConstants.CONVERSION_MINMAX;
        }
    }
    if(conversionComboBox.getSelectedIndex() == 2) {
        return select = ITagConstants.CONVERSION_FOMULA;
    }
    if(conversionComboBox.getSelectedIndex() == 3) {
        return select = ITagConstants.CONVERSION_SCALE_OFFSET;
    }
    return select;
}

private BigDecimal getDecimalValue(JTextField comt) throws FieldValueIncorrectException {
    BigDecimal decimalValue = BigDecimal.ZERO;

private void checkOverValue(Double parseDouble, JTextField comt) throws FieldValueIncorrectException {
    BigDecimal min = BigDecimal.ZERO;
    BigDecimal max = BigDecimal.valueOf(Double.MAX_VALUE);
    BigDecimal parse = BigDecimal.valueOf(parseDouble);
    if((min.doubleValue() > parse.doubleValue()) || (max.doubleValue() < parseDouble.doubleValue())) {
        throw FieldValueIncorrectException.fire(comt, Messages.getString("DevicePanel.WrongValue")); //$NON-NLS-1$
    }
}


private boolean isChangedString(String oldName, String newName) {
    boolean check = false;
    if(oldName == null) {
        if(newName == null) {
            return false;
        }
        if(newName.length() == 0) {
            check = false;
        }
        else {
            check = true;
        }
    }
    else {
        check = (!StringUtil.equals(oldName, newName));
    }
    return check;
}
