1483  public void arcTo(double rx, double ry,
1484  double xAxisRotation,
1485  boolean largeArcFlag, boolean sweepFlag,
1486  double x, double y
1487  ) {
1488 
1489 
1490  // Ensure radii are valid
1491  if (rx == 0 || ry == 0) {
1492  lineTo(x, y);
1493  return;
1494  }
1495 
1496  // Get the current (x, y) coordinates of the path
1497  Node lastPoint = get(size() - 1);
1498  double x0 = ((lastPoint.mask & C2_MASK) == C2_MASK) ? lastPoint.x[2] : lastPoint.x[0];
1499  double y0 = ((lastPoint.mask & C2_MASK) == C2_MASK) ? lastPoint.y[2] : lastPoint.y[0];
1500 
1501  if (x0 == x && y0 == y) {
1502  // If the endpoints (x, y) and (x0, y0) are identical, then this
1503  // is equivalent to omitting the elliptical arc segment entirely.
1504  return;
1505  }
1506 
1507  // Compute the half distance between the current and the final point
1508  double dx2 = (x0 - x) / 2d;
1509  double dy2 = (y0 - y) / 2d;
1510  // Convert angle from degrees to radians
1511  double angle = Math.toRadians(xAxisRotation);
1512  double cosAngle = Math.cos(angle);
1513  double sinAngle = Math.sin(angle);
1514 
1515  //
1516  // Step 1 : Compute (x1, y1)
1517  //
1518  double x1 = (cosAngle * dx2 + sinAngle * dy2);
1519  double y1 = (-sinAngle * dx2 + cosAngle * dy2);
1520  // Ensure radii are large enough
1521  rx = Math.abs(rx);
1522  ry = Math.abs(ry);
1523  double Prx = rx * rx;
1524  double Pry = ry * ry;
1525  double Px1 = x1 * x1;
1526  double Py1 = y1 * y1;
1527  // check that radii are large enough
1528  double radiiCheck = Px1/Prx + Py1/Pry;
1529  if (radiiCheck > 1) {
1530  rx = Math.sqrt(radiiCheck) * rx;
1531  ry = Math.sqrt(radiiCheck) * ry;
1532  Prx = rx * rx;
1533  Pry = ry * ry;
1534  }
1535 
1536  //
1537  // Step 2 : Compute (cx1, cy1)
1538  //
1539  double sign = (largeArcFlag == sweepFlag) ? -1 : 1;
1540  double sq = ((Prx*Pry)-(Prx*Py1)-(Pry*Px1)) / ((Prx*Py1)+(Pry*Px1));
1541  sq = (sq < 0) ? 0 : sq;
1542  double coef = (sign * Math.sqrt(sq));
1543  double cx1 = coef * ((rx * y1) / ry);
1544  double cy1 = coef * -((ry * x1) / rx);
1545 
1546  //
1547  // Step 3 : Compute (cx, cy) from (cx1, cy1)
1548  //
1549  double sx2 = (x0 + x) / 2.0;
1550  double sy2 = (y0 + y) / 2.0;
1551  double cx = sx2 + (cosAngle * cx1 - sinAngle * cy1);
1552  double cy = sy2 + (sinAngle * cx1 + cosAngle * cy1);
1553 
1554  //
1555  // Step 4 : Compute the angleStart (angle1) and the angleExtent (dangle)
1556  //
1557  double ux = (x1 - cx1) / rx;
1558  double uy = (y1 - cy1) / ry;
1559  double vx = (-x1 - cx1) / rx;
1560  double vy = (-y1 - cy1) / ry;
1561  double p, n;
1562 
1563  // Compute the angle start
1564  n = Math.sqrt((ux * ux) + (uy * uy));
1565  p = ux; // (1 * ux) + (0 * uy)
1566  sign = (uy < 0) ? -1d : 1d;
1567  double angleStart = Math.toDegrees(sign * Math.acos(p / n));
1568 
1569  // Compute the angle extent
1570  n = Math.sqrt((ux * ux + uy * uy) * (vx * vx + vy * vy));
1571  p = ux * vx + uy * vy;
1572  sign = (ux * vy - uy * vx < 0) ? -1d : 1d;
1573  double angleExtent = Math.toDegrees(sign * Math.acos(p / n));
1574  if(!sweepFlag && angleExtent > 0) {
1575  angleExtent -= 360f;
1576  } else if (sweepFlag && angleExtent < 0) {
1577  angleExtent += 360f;
1578  }
1579  angleExtent %= 360f;
1580  angleStart %= 360f;
1581 
1582  //
1583  // We can now build the resulting Arc2D in double precision
1584  //
1585  Arc2D.Double arc = new Arc2D.Double(
1586  cx - rx, cy - ry,
1587  rx * 2d, ry * 2d,
1588  -angleStart, -angleExtent,
1589  Arc2D.OPEN
1590  );
1591 
1592  // Create a path iterator of the rotated arc
1593  PathIterator i = arc.getPathIterator(
1594  AffineTransform.getRotateInstance(
1595  angle, arc.getCenterX(), arc.getCenterY()
1596  )
1597  );
1598 
1599  // Add the segments to the bezier path
1600  double[] coords = new double[6];
1601  i.next(); // skip first moveto
1602  while (! i.isDone()) {
1603  int type = i.currentSegment(coords);
1604  switch (type) {
1605  case PathIterator.SEG_CLOSE :
1606  // ignore
1607  break;
1608  case PathIterator.SEG_CUBICTO :
1609  curveTo(coords[0], coords[1], coords[2], coords[3], coords[4], coords[5]);
1610  break;
1611  case PathIterator.SEG_LINETO :
1612  lineTo(coords[0], coords[1]);

1614  case PathIterator.SEG_MOVETO :
1615  // ignore
1616  break;
1617  case PathIterator.SEG_QUADTO :

1619  break;
1620  }
1621  i.next();
1622  }
