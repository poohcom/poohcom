417  protected void writeAttributes(DOMOutput out) throws IOException {
418  Figure prototype = (Figure) out.getPrototype();
419 
420  boolean isElementOpen = false;
421  for (Map.Entry<AttributeKey, Object> entry : attributes.entrySet()) {
422  AttributeKey key = entry.getKey();
423  if (forbiddenAttributes == null
424  || ! forbiddenAttributes.contains(key)) {
425  Object prototypeValue = key.get(prototype);
426  Object attributeValue = key.get(this);
427  if (prototypeValue != attributeValue ||
428  (prototypeValue != null && attributeValue != null &&
429  ! prototypeValue.equals(attributeValue))) {
430  if (! isElementOpen) {
431  out.openElement("a");
432  isElementOpen = true;
433  }
434  out.openElement(key.getKey());
435  out.writeObject(entry.getValue());
436  out.closeElement();
437  }
438  }
439  }
440  if (isElementOpen) {
441  out.closeElement();
442  }
443  }

457  protected void readAttributes(DOMInput in) throws IOException {
458  if (in.getElementCount("a") > 0) {
459  in.openElement("a");
460  for (int i=in.getElementCount() - 1; i >= 0; i-- ) {
461  in.openElement(i);
462  String name = in.getTagName();
463  Object value = in.readObject();
464  AttributeKey key = getAttributeKey(name);
465  if (key != null && key.isAssignable(value)) {

468  setAttribute(key, value);
