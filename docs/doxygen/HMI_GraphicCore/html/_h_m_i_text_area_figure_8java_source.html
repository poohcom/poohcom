385  protected void drawText(Graphics2D g) {
386  if (getText() != null || isEditable()) {
387  Font font = getFont();
388  boolean isUnderlined = FONT_UNDERLINE.get(this);
389  Insets2D.Double insets = getInsets();
390  Rectangle2D.Double textRect = new Rectangle2D.Double(
391  bounds.x + insets.left,
392  bounds.y + insets.top,
393  bounds.width - insets.left - insets.right,
394  bounds.height - insets.top - insets.bottom
395  );
396 
397  Rectangle2D.Double oriTextRect = geomPool.poolRect2D(textRect);
398  double rotation = ROTATE_THETA.get(this);
399  if (rotation != 0) {
400  AffineTransform at = AffineTransform.getRotateInstance(rotation,
401  textRect.getCenterX(), textRect.getCenterY());
402  Rectangle2D r = at.createTransformedShape(textRect).getBounds2D();
403  textRect.x = r.getX();
404  textRect.y = r.getY();
405  textRect.width = r.getWidth();
406  textRect.height = r.getHeight();
407  }
408 
409  float leftMargin = (float) textRect.x;
410  float rightMargin = (float) Math.max(leftMargin + 1, textRect.x + textRect.width);
411  float verticalPos = (float) textRect.y;
412  float maxVerticalPos = (float) (textRect.y + textRect.height);
413  if (leftMargin < rightMargin) {
414  //float tabWidth = (float) (getTabSize() * g.getFontMetrics(font).charWidth('m'));
415  float tabWidth = (float) (getTabSize() * font.getStringBounds("m", getFontRenderContext()).getWidth());
416  float[] tabStops = new float[(int) (textRect.width / tabWidth)];
417  for (int i=0; i < tabStops.length; i++) {
418  tabStops[i] = (float) (textRect.x + (int) (tabWidth * (i + 1)));
419  }
420 
421  if (getText() != null) {
422  Shape savedClipArea = g.getClip();
423  g.clip(oriTextRect);
424  g.rotate(rotation, textRect.getCenterX(), textRect.getCenterY());
425  String[] paragraphs = getText().split("\n");//Strings.split(getText(), '\n');
426  for (int i = 0; i < paragraphs.length; i++) {
427  if (paragraphs[i].length() == 0) paragraphs[i] = " ";
428  AttributedString as = new AttributedString(paragraphs[i]);
429  as.addAttribute(TextAttribute.FONT, font);
430  if (isUnderlined) {
431  as.addAttribute(TextAttribute.UNDERLINE, TextAttribute.UNDERLINE_LOW_ONE_PIXEL);
432  }
433  int tabCount = new StringTokenizer(paragraphs[i], "\t").countTokens() - 1;
434  verticalPos = drawParagraph(g, as.getIterator(), verticalPos, maxVerticalPos, leftMargin, rightMargin, tabStops, tabCount, textRect);
435  if (verticalPos > maxVerticalPos) {
436  break;
437  }
438  }
439  g.rotate(-rotation, textRect.getCenterX(), textRect.getCenterY());
440  g.setClip(savedClipArea);
441  }
442  }
443  geomPool.offerRect2D(oriTextRect);
444  }

475  private float drawParagraph(Graphics2D g, AttributedCharacterIterator styledText,
476  float verticalPos, float maxVerticalPos, float leftMargin, float rightMargin,
477  float[] tabStops, int tabCount, Rectangle2D.Double textRect) {
478  // This method is based on the code sample given
479  // in the class comment of java.awt.font.LineBreakMeasurer,
480 
481  // assume styledText is an AttributedCharacterIterator, and the number
482  // of tabs in styledText is tabCount
483 
484  int[] tabLocations = new int[tabCount+1];
485 
486  int i = 0;
487  for (char c = styledText.first(); c != styledText.DONE; c = styledText.next()) {
488  if (c == '\t') {
489  tabLocations[i++] = styledText.getIndex();
490  }
491  }
492  tabLocations[tabCount] = styledText.getEndIndex() - 1;
493 
494  // Now tabLocations has an entry for every tab's offset in
495  // the text. For convenience, the last entry is tabLocations
496  // is the offset of the last character in the text.
497 
498  LineBreakMeasurer measurer = new LineBreakMeasurer(styledText, getFontRenderContext());
499  int currentTab = 0;
500 
501  while (measurer.getPosition() < styledText.getEndIndex() &&
502  verticalPos <= maxVerticalPos) {
503 
504  // Lay out and draw each line. All segments on a line
505  // must be computed before any drawing can occur, since
506  // we must know the largest ascent on the line.
507  // TextLayouts are computed and stored in a List;
508  // their horizontal positions are stored in a parallel
509  // List.
510 
511  // lineContainsText is true after first segment is drawn
512  boolean lineContainsText = false;
513  boolean lineComplete = false;
514  float maxAscent = 0, maxDescent = 0;
515  float horizontalPos = leftMargin;
516  LinkedList<TextLayout> layouts = new LinkedList<TextLayout>();
517  LinkedList<Float> penPositions = new LinkedList<Float>();
518 
519  while (!lineComplete && verticalPos <= maxVerticalPos) {
520  float wrappingWidth = rightMargin - horizontalPos;
521  TextLayout layout = null;
522  layout =
523  measurer.nextLayout(wrappingWidth,
524  tabLocations[currentTab]+1,
525  lineContainsText);
526 
527  // layout can be null if lineContainsText is true
528  if (layout != null) {
529  layouts.add(layout);
530  // 2009. 10. 13 수정
531  // 문자열 수평 정렬에 따라 위치를 지정한다.
532  HorizontalAlignment align = HORIZONTAL_ALIGN.get(this);
533  if (align == HorizontalAlignment.CENTER) {
534  horizontalPos = (float)(textRect.x + (textRect.width - layout.getBounds().getWidth())/2);
535  }
536  else if (align == HorizontalAlignment.RIGHT){
537  horizontalPos = (float)(textRect.x + (textRect.width - layout.getBounds().getWidth()) - 2);
538  }
539  penPositions.add(horizontalPos);
540  horizontalPos += layout.getAdvance();
541  maxAscent = Math.max(maxAscent, layout.getAscent());
542  maxDescent = Math.max(maxDescent,
543  layout.getDescent() + layout.getLeading());
544  } else {
545  lineComplete = true;
546  }
547 
548  lineContainsText = true;
549 
550  if (measurer.getPosition() == tabLocations[currentTab]+1) {
551  currentTab++;
552  }
553 
554  if (measurer.getPosition() == styledText.getEndIndex())
555  lineComplete = true;
556  else if (tabStops.length == 0 || horizontalPos >= tabStops[tabStops.length-1])
557  lineComplete = true;
558 
559  if (!lineComplete) {
560  // move to next tab stop
561  int j;
562  for (j=0; horizontalPos >= tabStops[j]; j++) {}
563  horizontalPos = tabStops[j];
564  }
565  }
566 
567  verticalPos += maxAscent;
568 
569  Iterator<TextLayout> layoutEnum = layouts.iterator();
570  Iterator<Float> positionEnum = penPositions.iterator();
571 
572  // now iterate through layouts and draw them
573  while (layoutEnum.hasNext()) {
574  TextLayout nextLayout = layoutEnum.next();
575  float nextPosition = positionEnum.next();
576  if (g != null) {
577  nextLayout.draw(g, nextPosition, verticalPos);
578  }
579  }
580 
581  verticalPos += maxDescent;
582  }
583 
584  return verticalPos;
585  }
