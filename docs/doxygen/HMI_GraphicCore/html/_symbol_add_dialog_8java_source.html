427  public void semanticCheck(Object obj) throws FieldValueIncorrectException {
428  if(null != library){
429  String symbolName = nameTextField.getText();
430 
431  if(symbolName.getBytes().length > 50){
432  throw FieldValueIncorrectException.fire(nameTextField, LibraryWindowResource.getString("Dialog.ErrorMsg.SymbolNameRange")); //$NON-NLS-1$
433  }
434 
435  boolean isUsePunctuation = false;
436 
437  for(char c : symbolName.toCharArray()){
438  int charType = Character.getType(c);
439  if(c != '_' && (charType >= Character.DASH_PUNCTUATION && charType <= Character.FINAL_QUOTE_PUNCTUATION)){
440  isUsePunctuation = true;
441  break;
442  }
443  }
444 
445  if(isUsePunctuation){
446  throw FieldValueIncorrectException.fire(nameTextField, LibraryWindowResource.getString("Dialog.ErrorMsg.Symbol")); //$NON-NLS-1$
447  }
448 
449  String categoryName = categoryList.getSelectedValue().toString();
450  Category category = library.getCategory(categoryName);
451 
452  if(category.contains(symbolName)){
453  throw FieldValueIncorrectException.fire(nameTextField, LibraryWindowResource.getString("Dialog.MessageDialog.Message.ExistSymbol")); //$NON-NLS-1$
