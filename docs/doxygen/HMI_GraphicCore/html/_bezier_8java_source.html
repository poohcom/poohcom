174  private static void fitCubic(Point2D.Double[] d, int first, int last,
175  Point2D.Double tHat1, Point2D.Double tHat2, double error, GeneralPath bezierPath) {
176  Point2D.Double[] bezCurve; /*Control points of fitted Bezier curve*/
177  double[] u; /* Parameter values for point */
178  double[] uPrime; /* Improved parameter values */
179  double maxError; /* Maximum fitting error */
180  int[] splitPoint = new int[1]; /* Point to split point set at.
181  This is an array of size one, because we need it as an input/output parameter.
182  */
183  int nPts; /* Number of points in subset */
184  double iterationError; /*Error below which you try iterating */
185  int maxIterations = 4; /* Max times to try iterating */
186  Point2D.Double tHatCenter = new Point2D.Double(); /* Unit tangent vector at splitPoint */
187  int i;
188 
189  iterationError = error * error;
190  nPts = last - first + 1;
191 
192  /* Use heuristic if region only has two points in it */
193  if (nPts == 2) {
194  double dist = v2DistanceBetween2Points(d[last], d[first]) / 3.0;
195 
196  bezCurve = new Point2D.Double[4];
197  for (i=0; i < bezCurve.length; i++) {
198  bezCurve[i] = new Point2D.Double();
199  }
200  bezCurve[0] = d[first];
201  bezCurve[3] = d[last];
202  v2Add(bezCurve[0], v2Scale(tHat1, dist), bezCurve[1]);
203  v2Add(bezCurve[3], v2Scale(tHat2, dist), bezCurve[2]);
204  bezierPath.curveTo(
205  (float) bezCurve[1].x, (float) bezCurve[1].y,
206  (float) bezCurve[2].x, (float) bezCurve[2].y,
207  (float) bezCurve[3].x, (float) bezCurve[3].y
208  );
209  return;
210  }
211 
212  /* Parameterize points, and attempt to fit curve */
213  u = chordLengthParameterize(d, first, last);
214  bezCurve = generateBezier(d, first, last, u, tHat1, tHat2);
215 
216  /* Find max deviation of points to fitted curve */
217  maxError = computeMaxError(d, first, last, bezCurve, u, splitPoint);
218  if (maxError < error) {
219  bezierPath.curveTo(
220  (float) bezCurve[1].x, (float) bezCurve[1].y,
221  (float) bezCurve[2].x, (float) bezCurve[2].y,
222  (float) bezCurve[3].x, (float) bezCurve[3].y
223  );
224  return;
225  }
226 
227 
228  /* If error not too large, try some reparameterization */
229  /* and iteration */
230  if (maxError < iterationError) {
231  for (i = 0; i < maxIterations; i++) {
232  uPrime = reparameterize(d, first, last, u, bezCurve);
233  bezCurve = generateBezier(d, first, last, uPrime, tHat1, tHat2);
234  maxError = computeMaxError(d, first, last, bezCurve, uPrime, splitPoint);
235  if (maxError < error) {
236  bezierPath.curveTo(
237  (float) bezCurve[1].x, (float) bezCurve[1].y,
238  (float) bezCurve[2].x, (float) bezCurve[2].y,
239  (float) bezCurve[3].x, (float) bezCurve[3].y
240  );
241  return;
242  }
243  u = uPrime;
244  }
245  }
246 
247  /* Fitting failed -- split at max error point and fit recursively */
248  tHatCenter = computeCenterTangent(d, splitPoint[0]);
249  fitCubic(d, first, splitPoint[0], tHat1, tHatCenter, error, bezierPath);
250  v2Negate(tHatCenter);
251  fitCubic(d, splitPoint[0], last, tHatCenter, tHat2, error, bezierPath);
252  }

377  private static Point2D.Double[] generateBezier(Point2D.Double[] d, int first, int last, double[] uPrime, Point2D.Double tHat1, Point2D.Double tHat2) {
378  int i;
379  Point2D.Double[][] A = new Point2D.Double[MAXPOINTS][2]; /* Precomputed rhs for eqn */
380  int nPts; /* Number of pts in sub-curve */
381  double[][] C = new double[2][2];/* Matrix C */
382  double[] X = new double[2]; /* Matrix X */
383  double det_C0_C1, /* Determinants of matrices */
384  det_C0_X,
385  det_X_C1;
386  double alpha_l, /* Alpha values, left and right */
387  alpha_r;
388  Point2D.Double tmp = new Point2D.Double(); /* Utility variable */
389  Point2D.Double[] bezCurve; /* RETURN bezier curve ctl pts */
390 
391  bezCurve = new Point2D.Double[4];
392  for (i=0; i < bezCurve.length; i++) {
393  bezCurve[i] = new Point2D.Double();
394  }
395 
396  nPts = last - first + 1;
397 
398 
399  /* Compute the A's */
400  for (i = 0; i < nPts; i++) {
401  Point2D.Double v1, v2;
402  v1 = (Point2D.Double) tHat1.clone();
403  v2 = (Point2D.Double) tHat2.clone();
404  v2Scale(v1, b1(uPrime[i]));
405  v2Scale(v2, b2(uPrime[i]));
406  A[i][0] = v1;
407  A[i][1] = v2;
408  }
409 
410  /* Create the C and X matrices */
411  C[0][0] = 0.0;
412  C[0][1] = 0.0;
413  C[1][0] = 0.0;
414  C[1][1] = 0.0;
415  X[0] = 0.0;
416  X[1] = 0.0;
417 
418  for (i = 0; i < nPts; i++) {
419  C[0][0] += v2Dot(A[i][0], A[i][0]);
420  C[0][1] += v2Dot(A[i][0], A[i][1]);
421  /* C[1][0] += V2Dot(&A[i][0], &A[i][1]);*/
422  C[1][0] = C[0][1];
423  C[1][1] += v2Dot(A[i][1], A[i][1]);
424 
425  tmp = v2SubII(d[first + i],
426  v2AddII(
427  v2ScaleIII(d[first], b0(uPrime[i])),
428  v2AddII(
429  v2ScaleIII(d[first], b1(uPrime[i])),
430  v2AddII(
431  v2ScaleIII(d[last], b2(uPrime[i])),
432  v2ScaleIII(d[last], b3(uPrime[i]))))));
433 
434 
435  X[0] += v2Dot(A[i][0], tmp);
436  X[1] += v2Dot(A[i][1], tmp);
437  }
438 
439  /* Compute the determinants of C and X */
440  det_C0_C1 = C[0][0] * C[1][1] - C[1][0] * C[0][1];
441  det_C0_X = C[0][0] * X[1] - C[0][1] * X[0];
442  det_X_C1 = X[0] * C[1][1] - X[1] * C[0][1];
443 
444  /* Finally, derive alpha values */
445  if (det_C0_C1 == 0.0) {
446  det_C0_C1 = (C[0][0] * C[1][1]) * 10e-12;
447  }
448  alpha_l = det_X_C1 / det_C0_C1;
449  alpha_r = det_C0_X / det_C0_C1;
450 
451 
452  /* If alpha negative, use the Wu/Barsky heuristic (see text) */
453  /* (if alpha is 0, you get coincident control points that lead to
454  * divide by zero in any subsequent NewtonRaphsonRootFind() call. */
455  if (alpha_l < 1.0e-6 || alpha_r < 1.0e-6) {
456  double dist = v2DistanceBetween2Points(d[last], d[first]) / 3.0;
457 
458  bezCurve[0] = d[first];
459  bezCurve[3] = d[last];
460  v2Add(bezCurve[0], v2Scale(tHat1, dist), bezCurve[1]);
461  v2Add(bezCurve[3], v2Scale(tHat2, dist), bezCurve[2]);
462  return (bezCurve);
463  }
464 
465  /* First and last control points of the Bezier curve are */
466  /* positioned exactly at the first and last data points */
467  /* Control points 1 and 2 are positioned an alpha distance out */
468  /* on the tangent vectors, left and right, respectively */
469  bezCurve[0] = d[first];
470  bezCurve[3] = d[last];
471  v2Add(bezCurve[0], v2Scale(tHat1, alpha_l), bezCurve[1]);
472  v2Add(bezCurve[3], v2Scale(tHat2, alpha_r), bezCurve[2]);
473  return (bezCurve);
474  }

526  private static double newtonRaphsonRootFind(Point2D.Double[] Q, Point2D.Double P, double u) {
527  double numerator, denominator;
528  Point2D.Double[] Q1 = new Point2D.Double[3], Q2 = new Point2D.Double[2]; /* Q' and Q'' */
529  Point2D.Double Q_u = new Point2D.Double(), Q1_u = new Point2D.Double(), Q2_u = new Point2D.Double(); /*u evaluated at Q, Q', & Q'' */
530  double uPrime; /* Improved u */
531  int i;
532 
533  /* Compute Q(u) */
534  Q_u = bezierII(3, Q, u);
535 
536  /* Generate control vertices for Q' */
537  for (i = 0; i <= 2; i++) {
538  Q1[i] = new Point2D.Double(
539  (Q[i+1].x - Q[i].x) * 3.0,
540  (Q[i+1].y - Q[i].y) * 3.0
541  );
542  }
543 
544  /* Generate control vertices for Q'' */
545  for (i = 0; i <= 1; i++) {
546  Q2[i] = new Point2D.Double(
547  (Q1[i+1].x - Q1[i].x) * 2.0,
548  (Q1[i+1].y - Q1[i].y) * 2.0
549  );
550  }
551 
552  /* Compute Q'(u) and Q''(u) */
553  Q1_u = bezierII(2, Q1, u);
554  Q2_u = bezierII(1, Q2, u);
555 
556  /* Compute f(u)/f'(u) */
557  numerator = (Q_u.x - P.x) * (Q1_u.x) + (Q_u.y - P.y) * (Q1_u.y);
558  denominator = (Q1_u.x) * (Q1_u.x) + (Q1_u.y) * (Q1_u.y) +
559  (Q_u.x - P.x) * (Q2_u.x) + (Q_u.y - P.y) * (Q2_u.y);
560 
561  /* u = u - f(u)/f'(u) */
562  uPrime = u - (numerator/denominator);
563  return (uPrime);
564  }

583  private static Point2D.Double bezierII(int degree, Point2D.Double[] V, double t) {
584  int i, j;
585  Point2D.Double Q; /* Point on curve at parameter t */
586  Point2D.Double[] Vtemp; /* Local copy of control points */
587 
588  /* Copy array */
589  Vtemp = new Point2D.Double[degree+1];
590  for (i = 0; i <= degree; i++) {
591  Vtemp[i] = (Point2D.Double) V[i].clone();
592  }
593 
594  /* Triangle computation */
595  for (i = 1; i <= degree; i++) {
596  for (j = 0; j <= degree-i; j++) {
597  Vtemp[j].x = (1.0 - t) * Vtemp[j].x + t * Vtemp[j+1].x;
598  Vtemp[j].y = (1.0 - t) * Vtemp[j].y + t * Vtemp[j+1].y;
599  }
600  }
601 
602  Q = Vtemp[0];
603  return Q;
604  }

804  Point2D.Double v = new Point2D.Double(); /* Vector from point to curve */
805

807  maxDist = 0.0;
808  for (i = first + 1; i < last; i++) {

813  maxDist = dist;
