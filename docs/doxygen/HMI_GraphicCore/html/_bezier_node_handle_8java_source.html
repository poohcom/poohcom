107  public void draw(Graphics2D g) {
108  BezierFigure f = getOwner();
109  int size = f.getNodeCount();
110  boolean isClosed = f.isClosed();
111  if (size > index) {
112  BezierPath.Node v = f.getNode(index);
113  if (v.mask == 0 ||
114  index == 0 && v.mask == BezierPath.C1_MASK && ! isClosed ||
115  index == size - 1 && v.mask == BezierPath.C2_MASK && ! isClosed
116  ) {
117  drawRectangle(g, HANDLE_FILL_COLOR, HANDLE_STROKE_COLOR);
118  } else if (v.mask == BezierPath.C1_MASK ||
119  v.mask == BezierPath.C2_MASK ||
120  index == 0 && ! isClosed ||
121  index == size - 1 && ! isClosed) {
122  drawDiamond(g, HANDLE_FILL_COLOR, HANDLE_STROKE_COLOR);
123  } else {

253  public void trackEnd(Point anchor, Point lead, int modifiersEx) {
254  BezierFigure f = getOwner();
255 
256  // Change node type
257  if ((modifiersEx & (InputEvent.META_DOWN_MASK | InputEvent.CTRL_DOWN_MASK | InputEvent.ALT_DOWN_MASK | InputEvent.SHIFT_DOWN_MASK)) != 0 &&
258  (modifiersEx & InputEvent.BUTTON2_DOWN_MASK) == 0) {
259  f.willChange();
260  BezierPath.Node v = f.getNode(index);
261  if (index > 0 && index < f.getNodeCount() || f.isClosed()) {
262  v.mask = (v.mask + 3) % 4;
263  } else if (index == 0) {
264  v.mask = ((v.mask & BezierPath.C2_MASK) == 0) ? BezierPath.C2_MASK : 0;
265  } else {
266  v.mask = ((v.mask & BezierPath.C1_MASK) == 0) ? BezierPath.C1_MASK : 0;
267  }
268  f.setNode(index, v);
269  f.changed();
270  fireHandleRequestSecondaryHandles();
271  }
272  view.getDrawing().fireUndoableEditHappened(new BezierNodeEdit(f,index,oldNode,f.getNode(index)));
273  view.getDrawing().fireUndoableEditHappened(edit);
274  }

344  public Collection<Handle> createSecondaryHandles() {
345  BezierFigure f = getOwner();
346  LinkedList<Handle> list = new LinkedList<Handle>();
347  BezierPath.Node v = f.getNode(index);
348  if ((v.mask & BezierPath.C1_MASK) != 0 &&
349  (index != 0 || f.isClosed())) {
350  list.add(new BezierControlPointHandle(f, index, 1, getTransformOwner()));
351  }
352  if ((v.mask & BezierPath.C2_MASK) != 0 &&
353  (index < f.getNodeCount() - 1 ||
354  f.isClosed())) {
355  list.add(new BezierControlPointHandle(f, index, 2, getTransformOwner()));
356  }
357  if (index > 0 || f.isClosed()) {
358  int i = (index == 0) ? f.getNodeCount() - 1 : index - 1;
359  v = f.getNode(i);
360  if ((v.mask & BezierPath.C2_MASK) != 0) {
361  list.add(new BezierControlPointHandle(f, i, 2, getTransformOwner()));
362  }
363  }
364  if (index < f.getNodeCount() - 1 || f.isClosed()) {
365  int i = (index == f.getNodeCount() - 1) ? 0 : index + 1;
366  v = f.getNode(i);
367  if ((v.mask & BezierPath.C1_MASK) != 0) {
368  list.add(new BezierControlPointHandle(f, i, 1, getTransformOwner()));
369  }
370  }
371  return list;
