462  private void syntaxCheck() throws BeanValueIncorrectExeception {
463  if(useSystemDbCheckBox.isSelected()) {
464  if(null == dbNameComboBox.getSelectedItem()){
465  throw BeanValueIncorrectExeception.fire(UmsConstants.getString("Customizer.ErrMsg.SystemDb")); //$NON-NLS-1$
466  }
467  }
468  else {
469  String sidName = sidTextField.getText();
470  int portNumber = (Integer) portSpinner.getValue();
471  String address = ipAddressTextField.getText();
472 
473  boolean isSidEmpty = sidName.trim().isEmpty();
474  boolean isSidBlank = sidName.contains(" "); //$NON-NLS-1$
475  boolean port = portNumber > 65535 || portNumber < 1;
476  boolean isAddress = (null == address || address.equals("0.0.0.0") || address.isEmpty()); //$NON-NLS-1$
477 
478  boolean isAreaCode = (areaCodeComboBox.getSelectedIndex() == 0);
479 
480  boolean errCheck = port || isSidEmpty || isSidBlank || isAddress || isAreaCode;
481 
482  while(errCheck){
483  if(port){
484  throw BeanValueIncorrectExeception.fire(UmsConstants.getString("Customizer.ErrMsg.Port")); //$NON-NLS-1$
485  }
486  if(isSidEmpty){
487  throw BeanValueIncorrectExeception.fire(UmsConstants.getString("Customizer.ErrMsg.SID.Empty")); //$NON-NLS-1$
488  }
489  if(isSidBlank){
490  throw BeanValueIncorrectExeception.fire(UmsConstants.getString("Customizer.ErrMsg.SID.Blank")); //$NON-NLS-1$
491  }
492  if(isAddress){
493  throw BeanValueIncorrectExeception.fire(UmsConstants.getString("Customizer.ErrMsg.IP.Empty")); //$NON-NLS-1$
494  }
495  if(isAreaCode) {
496  throw BeanValueIncorrectExeception.fire("권역 코드를 선택해주세요.");
497  }
498  }
499  }
500  }
