357  protected void transform(Point2D.Double anchor, Point2D.Double lead) {
358  Figure f = getOwner();
359  f.willChange();
360  Rectangle2D.Double oldBounds = getTransformedBounds();
361  Rectangle2D.Double newBounds = new Rectangle2D.Double(
362  Math.min(anchor.x, lead.x),
363  Math.min(anchor.y, lead.y),
364  Math.abs(anchor.x - lead.x),
365  Math.abs(anchor.y - lead.y)
366  );
367  double sx = newBounds.width / oldBounds.width;
368  double sy = newBounds.height / oldBounds.height;
369 
370  AffineTransform tx = new AffineTransform();
371  tx.translate(-oldBounds.x, -oldBounds.y);
372  if (! Double.isNaN(sx) && ! Double.isNaN(sy) &&
373  (sx != 1d || sy != 1d) &&
374  ! (sx < 0.0001) && ! (sy < 0.0001)) {
375  f.transform(tx);
376  tx.setToIdentity();
377  tx.scale(sx, sy);
378  f.transform(tx);
379  tx.setToIdentity();
380  }
381  tx.translate(newBounds.x, newBounds.y);
382  f.transform(tx);
383  f.changed();
384  }
