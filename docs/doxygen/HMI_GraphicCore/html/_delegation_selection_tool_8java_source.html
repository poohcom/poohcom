246  protected void showPopupMenu(Figure figure, Point p, Component c) {
247  if (DEBUG) System.out.println("DelegationSelectionTool.showPopupMenu "+figure);
248  JPopupMenu menu = new JPopupMenu();
249  popupMenu = menu;
250  JMenu submenu = null;
251  String submenuName = null;
252  LinkedList<Action> popupActions = new LinkedList<Action>();
253  if (figure != null) {
254  LinkedList<Action> figureActions = new LinkedList<Action>(
255  figure.getActions(viewToDrawing(p))
256  );
257  if (popupActions.size() != 0 && figureActions.size() != 0) {
258  popupActions.add(null);
259  }
260  popupActions.addAll(figureActions);
261  if (popupActions.size() != 0 && selectionActions.size() != 0) {
262  popupActions.add(null);
263  }
264  popupActions.addAll(selectionActions);
265  }
266  if (popupActions.size() != 0 && drawingActions.size() != 0) {
267  popupActions.add(null);
268  }
269  popupActions.addAll(drawingActions);
270 
271  HashMap<Object,ButtonGroup> buttonGroups = new HashMap<Object,ButtonGroup>();
272  for (Action a : popupActions) {
273  if (a != null && a.getValue(Actions.SUBMENU_KEY) != null) {
274  if (submenuName == null || ! submenuName.equals(a.getValue(Actions.SUBMENU_KEY))) {
275  submenuName = (String) a.getValue(Actions.SUBMENU_KEY);
276  submenu = new JMenu(submenuName);
277  menu.add(submenu);
278  }
279  } else {
280  submenuName = null;
281  submenu = null;
282  }
283  if (a == null) {
284  if (submenu != null) submenu.addSeparator();
285  else menu.addSeparator();
286  } else {
287  AbstractButton button;
288 
289  if (a.getValue(Actions.BUTTON_GROUP_KEY) != null) {
290  ButtonGroup bg = buttonGroups.get(a.getValue(Actions.BUTTON_GROUP_KEY));
291  if (bg == null) {
292  bg = new ButtonGroup();
293  buttonGroups.put(a.getValue(Actions.BUTTON_GROUP_KEY), bg);
294  }
295  button = new JRadioButtonMenuItem(a);
296  bg.add(button);
297  button.setSelected(a.getValue(Actions.SELECTED_KEY) == Boolean.TRUE);
298  } else if (a.getValue(Actions.SELECTED_KEY) != null) {
299  button = new JCheckBoxMenuItem(a);
300  button.setSelected(a.getValue(Actions.SELECTED_KEY) == Boolean.TRUE);
301  } else {
302  button = new JMenuItem(a);
303  }
304 
305  if (submenu != null) submenu.add(button);
306  else menu.add(button);
307  }
308  }
309  menu.show(c, p.x, p.y);
310  }

323  protected void handleDoubleClick(MouseEvent evt) {
324  if (DEBUG) System.out.println("DelegationSelectionTool.handleDoubleClick "+evt);
325  DrawingView v = getView();
326  Point pos = new Point(evt.getX(), evt.getY());
327  Handle handle = v.findHandle(pos);
328  if (handle != null) {
329  if (DEBUG) System.out.println("DelegationSelectionTool.handleDoubleClick by handle");
330  handle.trackDoubleClick(pos, evt.getModifiersEx());
331  } else {
332  Point2D.Double p = viewToDrawing(pos);
333  Figure outerFigure = getView().findFigure(pos);
334  Figure figure = outerFigure;
335  if (figure != null) {
336  if (DEBUG) System.out.println("DelegationSelectionTool.handleDoubleClick by figure");
337  Tool figureTool = figure.getTool(p);
338  if (figureTool == null) {
339  figure = getDrawing().findFigureInside(p);
340  if (figure != null) {
341  figureTool = figure.getTool(p);
342  }
343  }
344  if (figureTool != null) {
345  setTracker(figureTool);
346  figureTool.mousePressed(evt);
347  } else {
348  if (outerFigure.handleMouseClick(p, evt, getView())) {
349  v.clearSelection();
350  v.addToSelection(outerFigure);
351  } else {
352  v.clearSelection();
353  v.addToSelection(outerFigure);
354  v.setHandleDetailLevel(v.getHandleDetailLevel() + 1);
355  }
356  }
357  }
358  }
359  evt.consume();
360  }
