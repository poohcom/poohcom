687  Facility facility = null;

692  facilityCode = facility.getFacilityCode();
693  if(null != facilityCode && facilityCode.startsWith(regionCode)) {
694  regionTagID = key;
695  break;
696  }
697  }
698 
699  if(null != regionTagID) {
700  String[] regionTagGroups = regionTagID.split("\\.");
701  this.regionName = regionTagGroups[0];
