336  lock.lock();

340  figureCount++;
341  }
342  }

351  boolean b = figureSet.remove(f);
352  lock.unlock();
353 
354  f.blink(false);

356  figureCount--;
357  }

361  blinkStatus = false;
362  }
363  }

367
