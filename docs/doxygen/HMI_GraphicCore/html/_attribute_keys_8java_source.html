548  public static Stroke getStroke(Figure f) {
549  double strokeWidth = STROKE_WIDTH.get(f);
550  float miterLimit = (float) getStrokeTotalMiterLimit(f);
551  double dashFactor = IS_STROKE_DASH_FACTOR.get(f) ? strokeWidth : 1d;
552  double[] ddashes = STROKE_DASHES.get(f);
553  float[] dashes = null;
554  if (ddashes != null) {
555  dashes = new float[ddashes.length];
556  for (int i=0; i < dashes.length; i++) {
557  dashes[i] = (float) (ddashes[i] * dashFactor);
558  }
559  }
560  switch (STROKE_TYPE.get(f)) {
561  case BASIC :
562  default :
563  if ((null == dashes) || (dashes.length == 1)) {
564  return new BasicStroke((float) strokeWidth, STROKE_CAP.get(f),
565  STROKE_JOIN.get(f));
566  }
567  else {
568  return new BasicStroke((float) strokeWidth,
569  STROKE_CAP.get(f),
570  STROKE_JOIN.get(f) ,
571  miterLimit,
572  dashes, (float) (STROKE_DASH_PHASE.get(f) * dashFactor));
573  }
574  //not reached
575  case DOUBLE :
576  return new DoubleStroke(
577  (float) (STROKE_INNER_WIDTH_FACTOR.get(f) * strokeWidth),
578  (float) strokeWidth,
579  STROKE_CAP.get(f),
580  STROKE_JOIN.get(f),
581  miterLimit,
582  dashes, (float) (STROKE_DASH_PHASE.get(f).floatValue() * dashFactor));
583  //not reached
584  }
585  }

646  break;
