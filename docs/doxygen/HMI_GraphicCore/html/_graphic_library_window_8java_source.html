758  private void drawSymbol(){

761 
762  switch (selectedSymbol.getType()) {

765  beanTool.setBeanClassName((String) data);

767  drawEditor.setTool(beanTool);
768  break;
769  case ILibraryConstants.TYPE_SVG:

772  SVGSymbolCreationTool.TYPE_TEXT);

774  drawEditor.setTool(svgTool);
775  break;
776  case ILibraryConstants.TYPE_IMAGE:
777  if (null != imageTool) {
778  imageTool.setImageSource(selectedSymbol.getUniqueName(), (BufferedImage) data);
779  }
780  drawEditor.setTool(imageTool);
781  break;
782  case ILibraryConstants.TYPE_GROUP:
783  if (null != groupTool) {
784  groupTool.setPrototypeData((FigureEffectSetData) data);

786  drawEditor.setTool(groupTool);
