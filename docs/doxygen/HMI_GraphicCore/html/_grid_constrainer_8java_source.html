253  Point2D.Double p0 = constrainPoint((Point2D.Double) p.clone());
254 
255  switch (dir) {
256  case NORTH:
257  case NORTH_WEST:
258  case NORTH_EAST:
259  if (p0.y < p.y) {
260  p.y = p0.y;
261  } else if (p0.y > p.y) {
262  p.y = p0.y - height;
263  }
264  break;
265  case SOUTH:

267  case SOUTH_EAST:

351  Point2D.Double p1 = constrainPoint(new Point2D.Double(r.x + r.width, r.y + r.height));

523  public void draw(Graphics2D g, DrawingView view) {
524  if (isVisible) {
525  AffineTransform t = view.getDrawingToViewTransform();
526  Rectangle viewBounds = g.getClipBounds();
527  Rectangle2D.Double bounds = view.viewToDrawing(viewBounds);
528 
529  Point2D.Double origin = constrainPoint(new Point2D.Double(bounds.x, bounds.y));
530  Point2D.Double point = new Point2D.Double();
531  Point2D.Double viewPoint = new Point2D.Double();
532 
533  // vertical grid lines are only drawn, if they are at least two
534  // pixels apart on the view coordinate system.
535  if (width * view.getScaleFactor() > 2) {
536  g.setColor(minorColor);
537  for (int i = (int) (origin.x / width), m = (int) ((origin.x + bounds.width) / width) + 1; i <= m; i++) {
538  g.setColor((i % majorGridSpacing == 0) ? majorColor : minorColor);
539 
540  point.x = width * i;
541  t.transform(point, viewPoint);
542  g.drawLine((int) viewPoint.x, (int) viewBounds.y,
543  (int) viewPoint.x, (int) (viewBounds.y + viewBounds.height));
544  }
545  } else if (width * majorGridSpacing * view.getScaleFactor() > 2) {
546  g.setColor(majorColor);
547  for (int i = (int) (origin.x / width), m = (int) ((origin.x + bounds.width) / width) + 1; i <= m; i++) {
548  if (i % majorGridSpacing == 0) {
549  point.x = width * i;
550  t.transform(point, viewPoint);
551  g.drawLine((int) viewPoint.x, (int) viewBounds.y,
552  (int) viewPoint.x, (int) (viewBounds.y + viewBounds.height));
553  }
554  }
555  }
556 
557  // horizontal grid lines are only drawn, if they are at least two
558  // pixels apart on the view coordinate system.
559  if (height * view.getScaleFactor() > 2) {
560  g.setColor(minorColor);
561  for (int i = (int) (origin.y / height), m = (int) ((origin.y + bounds.height) / height) + 1; i <= m; i++) {
562  g.setColor((i % majorGridSpacing == 0) ? majorColor : minorColor);
563 
564  point.y = height * i;
565  t.transform(point, viewPoint);
566  g.drawLine((int) viewBounds.x, (int) viewPoint.y,
567  (int) (viewBounds.x + viewBounds.width), (int) viewPoint.y);
568  }
569  } else if (height * majorGridSpacing * view.getScaleFactor() > 2) {
570  g.setColor(majorColor);
571  for (int i = (int) (origin.y / height), m = (int) ((origin.y + bounds.height) / height) + 1; i <= m; i++) {
572  if (i % majorGridSpacing == 0) {

589  public double constrainAngle(double angle) {

605  public double rotateAngle(double angle, RotationDirection dir) {
606  // Check parameters
607  if (dir == null) {
608  throw new IllegalArgumentException("dir must not be null");
609  }
610 
611  // Rotate into the specified direction by theta

613  switch (dir) {

617  case COUNTER_CLOCKWISE :
618  default:
