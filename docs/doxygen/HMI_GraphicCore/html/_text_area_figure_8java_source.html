105  protected void drawText(Graphics2D g) {
106  if (getText() != null || isEditable()) {
107  Font font = getFont();
108  boolean isUnderlined = FONT_UNDERLINE.get(this);
109  Insets2D.Double insets = getInsets();
110  Rectangle2D.Double textRect = new Rectangle2D.Double(
111  bounds.x + insets.left,
112  bounds.y + insets.top,
113  bounds.width - insets.left - insets.right,
114  bounds.height - insets.top - insets.bottom
115  );
116 
117  Rectangle2D.Double oriTextRect = geomPool.poolRect2D(textRect);
118  double rotation = ROTATE_THETA.get(this);
119  if (rotation != 0) {
120  AffineTransform at = AffineTransform.getRotateInstance(rotation,
121  textRect.getCenterX(), textRect.getCenterY());
122  Rectangle2D r = at.createTransformedShape(textRect).getBounds2D();
123  textRect.x = r.getX();
124  textRect.y = r.getY();
125  textRect.width = r.getWidth();
126  textRect.height = r.getHeight();
127  }
128 
129  float leftMargin = (float) textRect.x;
130  float rightMargin = (float) Math.max(leftMargin + 1, textRect.x + textRect.width);
131  float verticalPos = (float) textRect.y;
132  float maxVerticalPos = (float) (textRect.y + textRect.height);
133  if (leftMargin < rightMargin) {
134  //float tabWidth = (float) (getTabSize() * g.getFontMetrics(font).charWidth('m'));
135  float tabWidth = (float) (getTabSize() * font.getStringBounds("m", getFontRenderContext()).getWidth());
136  float[] tabStops = new float[(int) (textRect.width / tabWidth)];
137  for (int i=0; i < tabStops.length; i++) {
138  tabStops[i] = (float) (textRect.x + (int) (tabWidth * (i + 1)));
139  }
140 
141  if (getText() != null) {
142  Shape savedClipArea = g.getClip();
143  g.clip(oriTextRect);
144  g.rotate(rotation, textRect.getCenterX(), textRect.getCenterY());
145  String[] paragraphs = getText().split("\n");//Strings.split(getText(), '\n');
146  for (int i = 0; i < paragraphs.length; i++) {
147  if (paragraphs[i].length() == 0) paragraphs[i] = " ";
148  AttributedString as = new AttributedString(paragraphs[i]);
149  as.addAttribute(TextAttribute.FONT, font);
150  if (isUnderlined) {
151  as.addAttribute(TextAttribute.UNDERLINE, TextAttribute.UNDERLINE_LOW_ONE_PIXEL);
152  }
153  int tabCount = new StringTokenizer(paragraphs[i], "\t").countTokens() - 1;
154  verticalPos = drawParagraph(g, as.getIterator(), verticalPos, maxVerticalPos, leftMargin, rightMargin, tabStops, tabCount);
155  if (verticalPos > maxVerticalPos) {
156  break;
157  }
158  }
159  g.rotate(-rotation, textRect.getCenterX(), textRect.getCenterY());
160  g.setClip(savedClipArea);
161  }
162  }
163  geomPool.offerRect2D(oriTextRect);
164  }

194  private float drawParagraph(Graphics2D g, AttributedCharacterIterator styledText, float verticalPos, float maxVerticalPos, float leftMargin, float rightMargin, float[] tabStops, int tabCount) {
195  // This method is based on the code sample given
196  // in the class comment of java.awt.font.LineBreakMeasurer,
197 
198  // assume styledText is an AttributedCharacterIterator, and the number
199  // of tabs in styledText is tabCount
200 
201  int[] tabLocations = new int[tabCount+1];
202 
203  int i = 0;
204  for (char c = styledText.first(); c != styledText.DONE; c = styledText.next()) {
205  if (c == '\t') {
206  tabLocations[i++] = styledText.getIndex();
207  }
208  }
209  tabLocations[tabCount] = styledText.getEndIndex() - 1;
210 
211  // Now tabLocations has an entry for every tab's offset in
212  // the text. For convenience, the last entry is tabLocations
213  // is the offset of the last character in the text.
214 
215  LineBreakMeasurer measurer = new LineBreakMeasurer(styledText, getFontRenderContext());
216  int currentTab = 0;
217 
218  while (measurer.getPosition() < styledText.getEndIndex() &&
219  verticalPos <= maxVerticalPos) {
220 
221  // Lay out and draw each line. All segments on a line
222  // must be computed before any drawing can occur, since
223  // we must know the largest ascent on the line.
224  // TextLayouts are computed and stored in a List;
225  // their horizontal positions are stored in a parallel
226  // List.
227 
228  // lineContainsText is true after first segment is drawn
229  boolean lineContainsText = false;
230  boolean lineComplete = false;
231  float maxAscent = 0, maxDescent = 0;
232  float horizontalPos = leftMargin;
233  LinkedList<TextLayout> layouts = new LinkedList<TextLayout>();
234  LinkedList<Float> penPositions = new LinkedList<Float>();
235 
236  while (!lineComplete && verticalPos <= maxVerticalPos) {
237  float wrappingWidth = rightMargin - horizontalPos;
238  TextLayout layout = null;
239  layout =
240  measurer.nextLayout(wrappingWidth,
241  tabLocations[currentTab]+1,
242  lineContainsText);
243 
244  // layout can be null if lineContainsText is true
245  if (layout != null) {
246  layouts.add(layout);
247  penPositions.add(horizontalPos);
248  horizontalPos += layout.getAdvance();
249  maxAscent = Math.max(maxAscent, layout.getAscent());
250  maxDescent = Math.max(maxDescent,
251  layout.getDescent() + layout.getLeading());
252  } else {
253  lineComplete = true;
254  }
255 
256  lineContainsText = true;
257 
258  if (measurer.getPosition() == tabLocations[currentTab]+1) {
259  currentTab++;
260  }
261 
262  if (measurer.getPosition() == styledText.getEndIndex())
263  lineComplete = true;
264  else if (tabStops.length == 0 || horizontalPos >= tabStops[tabStops.length-1])
265  lineComplete = true;
266 
267  if (!lineComplete) {
268  // move to next tab stop
269  int j;
270  for (j=0; horizontalPos >= tabStops[j]; j++) {}
271  horizontalPos = tabStops[j];
272  }
273  }
274 
275  verticalPos += maxAscent;
276 
277  Iterator<TextLayout> layoutEnum = layouts.iterator();
278  Iterator<Float> positionEnum = penPositions.iterator();
279 
280  // now iterate through layouts and draw them
281  while (layoutEnum.hasNext()) {
282  TextLayout nextLayout = layoutEnum.next();
283  float nextPosition = positionEnum.next();
284  if (g != null) {
285  nextLayout.draw(g, nextPosition, verticalPos);
286  }
287  }
288 
289  verticalPos += maxDescent;
290  }
291 
292  return verticalPos;
293  }
