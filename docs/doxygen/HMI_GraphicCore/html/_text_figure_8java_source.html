152  protected TextLayout getTextLayout() {
153  if (textLayout == null) {
154  String text = getText();
155  if (text == null || text.length() == 0) {
156  text = " ";
157  }
158 
159  FontRenderContext frc = getFontRenderContext();
160  HashMap<TextAttribute,Object> textAttributes = new HashMap<TextAttribute,Object>();
161  textAttributes.put(TextAttribute.FONT, getFont());
162  if (FONT_UNDERLINE.get(this)) {
163  textAttributes.put(TextAttribute.UNDERLINE, TextAttribute.UNDERLINE_LOW_ONE_PIXEL);
164  }
165  textLayout = new TextLayout(text, textAttributes, frc);
166  }
167  return textLayout;
168  }

198  protected Rectangle2D.Double getFigureDrawingArea() {
199  if (getText() == null) {
200  return getBounds();
201  } else {
202  TextLayout layout = getTextLayout();
203  Rectangle2D.Double r = new Rectangle2D.Double(
204  origin.x, origin.y, layout.getAdvance(), layout.getAscent()
205  );
206  Rectangle2D lBounds = layout.getBounds();
207  if (! lBounds.isEmpty() && ! Double.isNaN(lBounds.getX())) {
208  r.add(new Rectangle2D.Double(
209  lBounds.getX()+origin.x,
210  (lBounds.getY()+origin.y+layout.getAscent()),
211  lBounds.getWidth(),
212  lBounds.getHeight()
213  ));
214  }
215  // grow by two pixels to take antialiasing into account
216  Geom.grow(r, 2d, 2d);
217  return r;
218  }
