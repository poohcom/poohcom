698  public boolean getObject(CanvasAlarmInfo info, BeansUndoProperties props) {
699  // 복제된 칼라필터.
700  CanvasAlarmFilter modelFilter = tableModel.getFilter();
701 
702  if(info.getColorFilter() == null) {
703  info.setColorFilter(getFilter());
704  return true;
705  }
706  boolean levelCheck = filter.getLevelFilter().checkSameFilter(
707  modelFilter.getLevelFilter().getTable());
708  boolean stateCheck = filter.getStateFilter().checkSameFilter(
709  modelFilter.getStateFilter().getTable());
710  boolean groupCheck = filter.getGroupFilter().checkSameFilter(
711  modelFilter.getGroupFilter().getTable());
712  boolean typeCheck = filter.getTypeFilter().checkSameFilter(
713  modelFilter.getTypeFilter().getTable());
714 
715  if(levelCheck
716  || stateCheck
717  || groupCheck
718  || typeCheck
719  || (filter.isShowColumnLine() != isShowColumnLine())
720  || (!filter.getSelectBackground().equals(getSelectBackColor()))
721  || (!filter.getSelectForeground().equals(getSelectForeColor()))
722  || (filter.isShowRecodeLine() != isShowRecodeLine())
723  || (!filter.getDataFont().equals(getDataFont()))
724  || (filter.getDataGap() != getDataHeight())
725  || (!filter.getBackGroudColor().equals(getBackgroundColor()))
726  || (filter.getSelectColorMode() != getSelectRadio())) {
727  info.setColorFilter(getFilter());
728  return true;
729  }
730  return false;
731  }
