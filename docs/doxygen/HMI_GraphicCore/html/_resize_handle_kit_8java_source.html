348  public void trackStart(Point anchor, int modifiersEx) {
349  geometry = getOwner().getTransformRestoreData();
350  Point location = getLocation();
351  dx = -anchor.x + location.x;
352  dy = -anchor.y + location.y;
353 
354  originBounds = getOwner().getBounds();
355  x1 = originBounds.getMinX();
356  y1 = originBounds.getMinY();
357  x2 = originBounds.getMaxX();
358  y2 = originBounds.getMaxY();
359 
360  anchorConstrainP = view.viewToDrawing(location);
361  currentConstrainP = geomPool.poolPoint2D(anchorConstrainP);
362 
363  geomPool.offerPoint(location);
364  }
365 
372  public void trackStep(Point anchor, Point lead, int modifiersEx) {
373  if (!isDragging) {
374  isDragging = true;
375  }
376  if (getOwner().isTransformable()) {
377  Point2D.Double p = view.viewToDrawing(new Point(lead.x + dx, lead.y + dy));
378  view.getConstrainer().constrainPoint(p);
379 
380  if (AttributeKeys.TRANSFORM.get(getOwner()) != null) {
381  try {
382  AttributeKeys.TRANSFORM.get(getOwner()).inverseTransform(p, p);
383  } catch (NoninvertibleTransformException ex) {
384  if (DEBUG) {
385  ex.printStackTrace();
386  }
387  }
388  }
389 
390  trackStepNormalized(p, modifiersEx);
391 
392  currentConstrainP.setLocation(p);
393  geomPool.offerPoint2D(p);
394  }
395  }
396 
403  public void trackEnd(Point anchor, Point lead, int modifiersEx) {
404  if (getOwner().isTransformable()) {
405  fireUndoableEditHappened(
406  new GeometryEdit(getOwner(), geometry, getOwner().getTransformRestoreData()));
407  }
408  isDragging = false;
409  geomPool.offerPoint2D(anchorConstrainP);
410  geomPool.offerPoint2D(currentConstrainP);
411  }

464  NorthEastHandle(Figure owner) {
465  super(owner, RelativeLocator.northEast(true));
466  }
467 
474  protected void trackStepNormalized(Point2D.Double p, int modifiersEx) {
475  Point2D.Double p1, p2;
476 
477  if ((p.x < x1 + 1) || (p.y > y2 -1)) {
478  return;
479  }
480 
481  double xRatio = (x1 - p.x) / (x1 - anchorConstrainP.x);
482  double yRatio = (y2 - p.y) / (y2 - anchorConstrainP.y);
483 
484  // Shift 연산시 동일 비율로 증감시킨다.
485  if ((modifiersEx & InputEvent.SHIFT_DOWN_MASK) != 0) {
486  if (xRatio >= yRatio) {
487  yRatio = xRatio;
488  }
489  else {
490  xRatio = yRatio;
491  }
492  }
493 
494  p1 = geomPool.poolPoint2D(x1, Math.min(y2 - 1, y2 - originBounds.height * yRatio));
495  p2 = geomPool.poolPoint2D(Math.max(x1 + 1, x1 + originBounds.width * xRatio), y2);
496 
497  setBounds(p1, p2);
498 
499  geomPool.offerPoint2D(p1);
500  geomPool.offerPoint2D(p2);
501  }
