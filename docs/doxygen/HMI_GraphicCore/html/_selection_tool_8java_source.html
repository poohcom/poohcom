231  public void mousePressed(MouseEvent evt) {
232  if (getView() != null && getView().isEnabled()) {
233  super.mousePressed(evt);
234  DrawingView view = getView();
235  Handle handle = view.findHandle(anchor);
236  Tool newTracker = null;
237  if (handle != null) {
238  newTracker = createHandleTracker(handle);
239  } else {
240  Figure figure;
241  if (isSelectBehindEnabled() &&
242  (evt.getModifiersEx() &
243  (InputEvent.ALT_DOWN_MASK | InputEvent.CTRL_DOWN_MASK)) != 0) {
244  // Select a figure behind the current selection
245  figure = view.findFigure(anchor);
246  HashSet<Figure> ignoredFigures = new HashSet<Figure>(view.getSelectedFigures());
247  ignoredFigures.add(figure);
248  Figure figureBehind = view.getDrawing().findFigureBehind(
249  view.viewToDrawing(anchor), ignoredFigures);
250  if (figureBehind != null) {
251  figure = figureBehind;
252  }
253  } else {
254  // If possible, continue to work with the current selection
255  Point2D.Double p = view.viewToDrawing(anchor);
256  figure = null;
257  if (isSelectBehindEnabled()) {
258  for (Figure f : view.getSelectedFigures()) {
259  if (f.contains(p)) {
260  figure = f;
261  break;
262  }
263  }
264  }
265  // If the point is not contained in the current selection,
266  // search for a figure.
267  if (figure == null) {
268  figure = view.findFigure(anchor);
269  }
270  }
271 
272  if (figure != null) {
273  newTracker = createDragTracker(figure);
274  } else {
275  if (! evt.isShiftDown()) {
276  view.clearSelection();
277  view.setHandleDetailLevel(0);
278  }
279  newTracker = createAreaTracker();
280  }
281  }
282 
283  if (newTracker != null) {
284  setTracker(newTracker);
285  }
286  tracker.mousePressed(evt);
287  }
288  }
