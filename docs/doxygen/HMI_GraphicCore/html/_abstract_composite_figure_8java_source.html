482  public void setBounds(Point2D.Double anchor, Point2D.Double lead) {
483  Rectangle2D.Double oldBounds = getBounds();
484  Rectangle2D.Double newBounds = new Rectangle2D.Double(
485  Math.min(anchor.x, lead.x),
486  Math.min(anchor.y, lead.y),
487  Math.abs(anchor.x - lead.x),
488  Math.abs(anchor.y - lead.y));
489 
490  double sx = newBounds.width / oldBounds.width;
491  double sy = newBounds.height / oldBounds.height;
492 
493  AffineTransform tx = new AffineTransform();
494  tx.translate(-oldBounds.x, -oldBounds.y);
495  if (!Double.isNaN(sx) && !Double.isNaN(sy) &&
496  (sx != 1d || sy != 1d) &&
497  !(sx < 0.0001) && !(sy < 0.0001)) {
498  transform(tx);
499  tx.setToIdentity();
500  tx.scale(sx, sy);
501  transform(tx);
502  tx.setToIdentity();
503  }
504  tx.translate(newBounds.x, newBounds.y);
505  transform(tx);
506  }

593  public boolean contains(Point2D.Double p) {
594  if (TRANSFORM.get(this) != null) {
595  try {
596  p = (Point2D.Double) TRANSFORM.get(this).inverseTransform(p, new Point2D.Double());
597  } catch (NoninvertibleTransformException ex) {
598  InternalError error = new InternalError(ex.getMessage());
599  error.initCause(ex);
600  throw error;
601  }
602  }
603  ;
604  if (getDrawingArea().contains(p)) {
605  for (Figure child : getChildrenFrontToBack()) {
606  if (child.isVisible() && child.contains(p)) {
607  return true;
608  }
609  }
610  }
611  return false;
