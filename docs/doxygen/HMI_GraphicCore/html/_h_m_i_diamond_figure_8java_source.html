187  @Override
188  public boolean contains(Double p) {
189  Rectangle2D.Double r = getBounds();
190  if(((Boolean)IS_QUADRATIC.get(this)).booleanValue())
191  {
192  double side = Math.max(r.width, r.height);
193  r.x -= (side - r.width) / 2D;
194  r.y -= (side - r.height) / 2D;
195  r.width = r.height = side;
196  }
197  double grow = AttributeKeys.getPerpendicularFillGrowth(this);
198  double growx;
199  double growy;
200  if(grow == 0.0D)
201  {
202  growx = growy = 0.0D;
203  } else
204  {
205  double w = r.width / 2D;
206  double h = r.height / 2D;
207  double lineLength = Math.sqrt(w * w + h * h);
208  double scale = grow / lineLength;
209  double yb = scale * w;
210  double xa = scale * h;
211  growx = (yb * yb) / xa + xa;
212  growy = (xa * xa) / yb + yb;
213  Geom.grow(r, growx, growy);
214  }
215  GeneralPath diamond = new GeneralPath();
216  diamond.moveTo((float)(r.x + r.width / 2D), (float)r.y);
217  diamond.lineTo((float)(r.x + r.width), (float)(r.y + r.height / 2D));
218  diamond.lineTo((float)(r.x + r.width / 2D), (float)(r.y + r.height));
219  diamond.lineTo((float)r.x, (float)(r.y + r.height / 2D));
220  diamond.closePath();
