356  protected void addPointToFigure(Point2D.Double newPoint) {
357  int pointCount = createdFigure.getNodeCount();
358 
359  createdFigure.willChange();
360  if (pointCount < 2) {
361  createdFigure.addNode(new BezierPath.Node(newPoint));
362  } else {
363  Point2D.Double endPoint = createdFigure.getEndPoint();
364  Point2D.Double secondLastPoint = (pointCount <= 1) ? endPoint : createdFigure.getPoint(pointCount - 2, 0);
365  if (newPoint.equals(endPoint)) {
366  createdFigure.setPoint(pointCount - 1, 0 , newPoint);
367  } else if (pointCount > 1 && Geom.lineContainsPoint(newPoint.x, newPoint.y, secondLastPoint.x, secondLastPoint.y, endPoint.x, endPoint.y, 0.9f / getView().getScaleFactor())) {
368  createdFigure.setPoint(pointCount - 1, 0, newPoint);
369  } else {
370  createdFigure.addNode(new BezierPath.Node(newPoint));
371  }
372  }
373  createdFigure.changed();
374  }

388  Rectangle r = new Rectangle(p);
389  r.grow(2, 2);
390  if (r.contains(evt.getX(), evt.getY())) {
391  createdFigure.setClosed(true);
392 
393  finishCreation(createdFigure);
394  createdFigure = null;
