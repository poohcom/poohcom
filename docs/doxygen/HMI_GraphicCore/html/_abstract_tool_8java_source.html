356  public void keyPressed(KeyEvent evt) {
357  if (! evt.isConsumed()) {
358  if (evt.getSource() instanceof Container) {
359  editor.setActiveView(editor.findView((Container) evt.getSource()));
360  }
361 
362  if (inputMap != null) {
363  Object obj = inputMap.get(KeyStroke.getKeyStroke(evt.getKeyCode(),evt.getModifiers(), false));
364  ActionListener al = null;
365  if (obj instanceof ActionListener) {
366  al = (ActionListener) obj;
367  } else if (obj != null && actionMap != null) {
368  al = actionMap.get(obj);
369  }
370  if (al != null) {
371  evt.consume();
372  al.actionPerformed(new ActionEvent(this, ActionEvent.ACTION_PERFORMED, "tool", evt.getWhen(), evt.getModifiers()));
373  }
374  }
375  }

664  public void updateCursor(DrawingView view, Point p) {
665  if (view.isEnabled()) {
666  Handle handle = view.findHandle(p);
667  if (handle != null) {
668  view.setCursor(handle.getCursor());
669  } else {
670  Figure figure = view.findFigure(p);
671  if (figure != null) {
672  Point2D.Double convertP = view.viewToDrawing(p);
673  view.setCursor(figure.getCursor(convertP));
674  geomPool.offerPoint2D(convertP);
675  } else {
676  view.setCursor(Cursor.getDefaultCursor());
677  }

679  } else {
