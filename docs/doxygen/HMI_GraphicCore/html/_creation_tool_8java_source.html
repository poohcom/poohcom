329  public void mouseDragged(MouseEvent evt) {
330  if (createdFigure != null) {
331  Point2D.Double constrainP = constrainPoint(evt.getPoint());
332 
333  if (!constrainP.equals(currentConstrainPos)) {
334  currentConstrainPos.setLocation(constrainP.x, constrainP.y);
335  createdFigure.willChange();
336  // Shift 키 상태일 경우에는 현재 좌표 중 anchor로 부터 더 멀리 있는 쪽을
337  // 기준으로 하여 직선 또는 정다각형을 만든다.
338  if (evt.isShiftDown()) {
339  boolean isXBase = Math.abs(constrainP.x - anchor.x) >= Math.abs(constrainP.y - anchor.y);
340  if (isLineCreate) {
341  if (isXBase) {
342  constrainP.setLocation(constrainP.x, anchor.y);
343  }
344  else {
345  constrainP.setLocation(anchor.x, constrainP.y);
346  }
347  }
348  else {
349  double size = (isXBase) ? (constrainP.x - anchor.x) : (constrainP.y - anchor.y);
350  constrainP.setLocation(anchor.x + size, anchor.y + size);
351  }
352  }

355  createdFigure.setBounds(constrainAnchor, constrainP);
