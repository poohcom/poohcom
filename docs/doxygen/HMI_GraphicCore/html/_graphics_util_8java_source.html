110  public static void drawImage(Graphics2D g2d,
111  CachableRed cr) {
112 
113  // System.out.println("DrawImage G: " + g2d);
114 
115  AffineTransform at = null;
116  while (true) {
117  if (cr instanceof AffineRed) {
118  AffineRed ar = (AffineRed)cr;
119  if (at == null)
120  at = ar.getTransform();
121  else
122  at.concatenate(ar.getTransform());
123  cr = ar.getSource();
124  continue;
125  } else if (cr instanceof TranslateRed) {
126  TranslateRed tr = (TranslateRed)cr;
127  // System.out.println("testing Translate");
128  int dx = tr.getDeltaX();
129  int dy = tr.getDeltaY();
130  if (at == null)
131  at = AffineTransform.getTranslateInstance(dx, dy);
132  else
133  at.translate(dx, dy);
134  cr = tr.getSource();
135  continue;
136  }
137  break;
138  }
139  AffineTransform g2dAt = g2d.getTransform();
140  if ((at == null) || (at.isIdentity()))
141  at = g2dAt;
142  else
143  at.preConcatenate(g2dAt);
144 
145  ColorModel srcCM = cr.getColorModel();
146  ColorModel g2dCM = getDestinationColorModel(g2d);
147  ColorSpace g2dCS = null;
148  if (g2dCM != null)
149  g2dCS = g2dCM.getColorSpace();
150  if (g2dCS == null)
151  // Assume device is sRGB
152  g2dCS = ColorSpace.getInstance(ColorSpace.CS_sRGB);
153 
154  ColorModel drawCM = g2dCM;
155  if ((g2dCM == null) || !g2dCM.hasAlpha()) {
156  // If we can't find out about our device or the device
157  // does not support alpha just use SRGB unpremultiplied
158  // (Just because this seems to work for us).
159  drawCM = sRGB_Unpre;
160  }
161 
162  if (cr instanceof BufferedImageCachableRed) {
163  // There is a huge win if we can use the BI directly here.
164  // This results in something like a 10x performance gain
165  // for images, the best thing is this is the common case.
166  if (g2dCS.equals(srcCM.getColorSpace()) &&
167  drawCM.equals(srcCM)) {
168  // System.err.println("Fast Case");
169  g2d.setTransform(at);
170  BufferedImageCachableRed bicr;
171  bicr = (BufferedImageCachableRed)cr;
172  g2d.drawImage(bicr.getBufferedImage(),
173  bicr.getMinX(), bicr.getMinY(), null);
174  g2d.setTransform(g2dAt);
175  return;
176  }
177  }
178 
179  // Scaling down so do it before color conversion.
180  double determinant = at.getDeterminant();
181  if (!at.isIdentity() && (determinant <= 1.0)) {
182  if (at.getType() != AffineTransform.TYPE_TRANSLATION)
183  cr = new AffineRed(cr, at, g2d.getRenderingHints());
184  else {
185  int xloc = cr.getMinX() + (int)at.getTranslateX();
186  int yloc = cr.getMinY() + (int)at.getTranslateY();
187  cr = new TranslateRed(cr, xloc, yloc);
188  }
189  }
190 
191  if (g2dCS != srcCM.getColorSpace()) {
192  // System.out.println("srcCS: " + srcCM.getColorSpace());
193  // System.out.println("g2dCS: " + g2dCS);
194  // System.out.println("sRGB: " +
195  // ColorSpace.getInstance(ColorSpace.CS_sRGB));
196  // System.out.println("LsRGB: " +
197  // ColorSpace.getInstance
198  // (ColorSpace.CS_LINEAR_RGB));
199  if (g2dCS == ColorSpace.getInstance(ColorSpace.CS_sRGB))
200  cr = convertTosRGB(cr);
201  else if (g2dCS == ColorSpace.getInstance(ColorSpace.CS_LINEAR_RGB))
202  cr = convertToLsRGB(cr);
203  }
204  srcCM = cr.getColorModel();
205  if (!drawCM.equals(srcCM))
206  cr = FormatRed.construct(cr, drawCM);
207 
208  // Scaling up so do it after color conversion.
209  if (!at.isIdentity() && (determinant > 1.0))
210  cr = new AffineRed(cr, at, g2d.getRenderingHints());
211 
212  // Now CR is in device space, so clear the g2d transform.
213  g2d.setTransform(IDENTITY);
214 
215  // Ugly Hack alert. This Makes it use our SrcOver implementation
216  // Which doesn't seem to have as many bugs as the JDK one when
217  // going between different src's and destinations (of course it's
218  // also a lot slower).
219  Composite g2dComposite = g2d.getComposite();
220  if (g2d.getRenderingHint(RenderingHintsKeyExt.KEY_TRANSCODING) ==
221  RenderingHintsKeyExt.VALUE_TRANSCODING_PRINTING) {
222  if (SVGComposite.OVER.equals(g2dComposite)) {
223  g2d.setComposite(SVGComposite.OVER);
224  }
225  }
226  Rectangle crR = cr.getBounds();
227  Shape clip = g2d.getClip();
228 
229  try {
230  Rectangle clipR;
231  if (clip == null) {
232  clip = crR;
233  clipR = crR;
234  } else {
235  clipR = clip.getBounds();
236 
237  if ( ! clipR.intersects(crR) )
238  return; // Nothing to draw...
239  clipR = clipR.intersection(crR);
240  }
241 
242  Rectangle gcR = getDestinationBounds(g2d);
243  // System.out.println("ClipRects: " + clipR + " -> " + gcR);
244  if (gcR != null) {
245  if ( ! clipR.intersects(gcR) )
246  return; // Nothing to draw...
247  clipR = clipR.intersection(gcR);
248  }
249 
250  // System.out.println("Starting Draw: " + cr);
251  // long startTime = System.currentTimeMillis();
252 
253  boolean useDrawRenderedImage = false;
254 
255  srcCM = cr.getColorModel();
256  SampleModel srcSM = cr.getSampleModel();
257  if ((srcSM.getWidth()*srcSM.getHeight()) >=
258  (clipR.width*clipR.height))
259  // if srcSM tiles are around the clip size
260  // then just draw the renderedImage
261  useDrawRenderedImage = true;
262 
263  Object atpHint = g2d.getRenderingHint
264  (RenderingHintsKeyExt.KEY_AVOID_TILE_PAINTING);
265 
266  if (atpHint == RenderingHintsKeyExt.VALUE_AVOID_TILE_PAINTING_ON)
267  useDrawRenderedImage = true; //for PDF and PS transcoders
268 
269  if (atpHint == RenderingHintsKeyExt.VALUE_AVOID_TILE_PAINTING_OFF)
270  useDrawRenderedImage = false;
271 
272 
273  WritableRaster wr;
274  if (useDrawRenderedImage) {
275  // This can be significantly faster but can also
276  // require much more memory, so we only use it when
277  // the clip size is smaller than the tile size.
278  Raster r = cr.getData(clipR);
279  wr = ((WritableRaster)r).createWritableChild
280  (clipR.x, clipR.y, clipR.width, clipR.height,
281  0, 0, null);
282 
283  BufferedImage bi = new BufferedImage
284  (srcCM, wr, srcCM.isAlphaPremultiplied(), null);
285 
286  // Any of the drawImage calls that take an
287  // Affine are prone to the 'CGGStackRestore: gstack
288  // underflow' bug on Mac OS X. This should work
289  // around that problem.
290  g2d.drawImage(bi, clipR.x, clipR.y, null);
291  } else {
292  // Use tiles to draw image...
293  wr = Raster.createWritableRaster(srcSM, new Point(0,0));
294  BufferedImage bi = new BufferedImage
295  (srcCM, wr, srcCM.isAlphaPremultiplied(), null);
296 
297  int xt0 = cr.getMinTileX();
298  int xt1 = xt0+cr.getNumXTiles();
299  int yt0 = cr.getMinTileY();
300  int yt1 = yt0+cr.getNumYTiles();
301  int tw = srcSM.getWidth();
302  int th = srcSM.getHeight();
303 
304  Rectangle tR = new Rectangle(0,0,tw,th);
305  Rectangle iR = new Rectangle(0,0,0,0);
306 
307  if (false) {
308  System.err.println("SrcCM: " + srcCM);
309  System.err.println("CR: " + cr);
310  System.err.println("CRR: " + crR + " TG: [" +
311  xt0 + ',' +
312  yt0 + ',' +
313  xt1 + ',' +
314  yt1 +"] Off: " +
315  cr.getTileGridXOffset() + ',' +
316  cr.getTileGridYOffset());
317  }
318 
319  int yloc = yt0*th+cr.getTileGridYOffset();
320  int skip = (clipR.y-yloc)/th;
321  if (skip <0) skip = 0;
322  yt0+=skip;
323 
324  int xloc = xt0*tw+cr.getTileGridXOffset();
325  skip = (clipR.x-xloc)/tw;
326  if (skip <0) skip = 0;
327  xt0+=skip;
328 
329  int endX = clipR.x+clipR.width-1;
330  int endY = clipR.y+clipR.height-1;
331 
332  if (false) {
333  System.out.println("clipR: " + clipR + " TG: [" +
334  xt0 + ',' +
335  yt0 + ',' +
336  xt1 + ',' +
337  yt1 +"] Off: " +
338  cr.getTileGridXOffset() + ',' +
339  cr.getTileGridYOffset());
340  }
341 
342 
343  yloc = yt0*th+cr.getTileGridYOffset();
344  int minX = xt0*tw+cr.getTileGridXOffset();
345  int xStep = tw;
346  xloc = minX;
347  for (int y=yt0; y<yt1; y++, yloc += th) {
348  if (yloc > endY) break;
349  for (int x=xt0; x<xt1; x++, xloc+=xStep) {
350  if ((xloc<minX) || (xloc > endX)) break;
351  tR.x = xloc;
352  tR.y = yloc;
353  Rectangle2D.intersect(crR, tR, iR);
354 
355  WritableRaster twr;
356  twr = wr.createWritableChild(0, 0,
357  iR.width, iR.height,
358  iR.x, iR.y, null);
359 
360  // System.out.println("Generating tile: " + twr);
361  cr.copyData(twr);
362 
363  // Make sure we only draw the region that was written.
364  BufferedImage subBI;
365  subBI = bi.getSubimage(0, 0, iR.width, iR.height);
366 
367  if (false) {
368  System.out.println("Drawing: " + tR);
369  System.out.println("IR: " + iR);
370  }
371 
372  // For some reason using the transform version
373  // causes a gStackUnderflow error but if I just
374  // use the drawImage with an x & y it works.
375  g2d.drawImage(subBI, iR.x, iR.y, null);
376  // AffineTransform trans
377  // = AffineTransform.getTranslateInstance(iR.x, iR.y);
378  // g2d.drawImage(subBI, trans, null);
379 
380  // String label = "sub [" + x + ", " + y + "]: ";
381  // org.ImageDisplay.showImage
382  // (label, subBI);
383  }
384  xStep = -xStep; // Reverse directions.
385  xloc += xStep; // Get back in bounds.
386  }
387  }
388  // long endTime = System.currentTimeMillis();
389  // System.out.println("Time: " + (endTime-startTime));
390 
391 
392  } finally {
393  g2d.setTransform(g2dAt);
394  g2d.setComposite(g2dComposite);
395  }
396 
397  // System.out.println("Finished Draw");
398  }
