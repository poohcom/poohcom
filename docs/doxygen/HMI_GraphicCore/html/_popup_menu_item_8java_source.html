126  public Action getAction(Viewer viewer) {
127  Action action = null;
128  switch (actionType) {
129  case ACTION_NONE:
130  break;
131  case ACTION_OPEN_WINDOW:
132  case ACTION_CLOSE_WINDOW:
133  case ACTION_OPEN_POPUP:
134  case ACTION_CLOSE_POPUP:
135  List<String> windows = new ArrayList<String> ();
136  if (null != actionTarget) {
137  String[] tokens = actionTarget.split(";");
138  for (String s : tokens) {
139  if (s.trim().length() > 0) {
140  windows.add(s);
141  }
142  }
143  }
144  if (!windows.isEmpty()) {
145  action = new WindowAction(viewer, actionType, windows);
146  }
147  break;
148  case ACTION_OPEN_URL:
149  try {
150  URL url = new URL(actionTarget);
151  action = new OpenUrlAction(viewer, url);
152  }
153  catch (Exception e) {
154  e.printStackTrace();
155  }
156  break;
157  case ACTION_FACILITY_SUMMARY:
158  // 시설물 코드는 18자리 문자열이어야 유효하다.
159  if ((null != actionTarget) && (actionTarget.length() == 18)) {
160  action = new ShowFacilitySummaryAction(viewer, actionTarget);
161  }
162  break;
163  case ACTION_SCRIPT:
164  if ((null != actionTarget) && !actionTarget.trim().isEmpty()) {
165  action = new ScriptExecuteAction(actionTarget);
166  }
167  break;
168  }
169 
170  if (null != action) {
171  this.viewer = viewer;
172  action.putValue(Action.NAME, menuName);
173  }
174
